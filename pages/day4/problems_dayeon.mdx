# 문제풀이(19 ~ 20)

## 문제 19
```java filename="Problem19.java" showLineNumbers copy
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

string solution(vector<string> participant, vector<string> completion) {
    // 참가자와 완주자 명단을 정렬
    sort(participant.begin(), participant.end());
    sort(completion.begin(), completion.end());

    // 참가자와 완주자를 비교하여 일치하지 않는 이름을 찾음
    for (int i = 0; i < completion.size(); i++) {
        if (participant[i] != completion[i]) {
            return participant[i];
        }
    }

    // 마지막 참가자가 완주하지 못한 경우
    return participant[participant.size() - 1];
}

```
## 문제 20
```java filename="Problem20.java" showLineNumbers copy
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>
#include <string.h>

int solution(const char* want[], size_t want_len, int number[], size_t number_len, const char* discount[], size_t discount_len) {
    int answer = 0;

    for (int start = 0; start <= discount_len - 10; start++) {
        int temp_count[want_len];
        for (int i = 0; i < want_len; i++) {
            temp_count[i] = number[i];
        }

        for (int day = start; day < start + 10; day++) {
            for (int i = 0; i < want_len; i++) {
                if (strcmp(discount[day], want[i]) == 0) {
                    temp_count[i]--;
                }
            }
        }

        bool allMatched = true;
        for (int i = 0; i < want_len; i++) {
            if (temp_count[i] > 0) {
                allMatched = false;
                break;
            }
        }

        if (allMatched) {
            answer++;
        }
    }

    return answer;
}
```
